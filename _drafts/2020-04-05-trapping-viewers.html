One day, a friend suggested me to add a back button re-directing to the home page on my pages. So naturally I go to google, and do a search


    function goBack() {
      const lastWebsite = document.referrer;
      const website = "cindyxmiao.github.io/blog/";
      const localHost = "http://localhost:4000/";
      if (lastWebsite.includes(website) | lastWebsite.includes(localHost)){
        window.history.back();
      }
      else{
        location.href='{{site.baseurl}}/';
      }
    }
